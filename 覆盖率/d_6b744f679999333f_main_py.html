32# 计算文章相似度 

33def calc_similarity(text1, text2): 

34 texts = [text1, text2] 

35 dictionary = gensim.corpora.Dictionary(texts) # 根据两篇文章的所有内容构建词表 

36 corpus = [dictionary.doc2bow(text) for text in texts] # 记录词表中所有词的频率 

37 similarities = gensim.similarities.Similarity(None, corpus=corpus, num_features=len(dictionary)) # 参数为：碎片文件名路径(不管,在缓存中随机读取）、词频、字典大小 

38 test_corpus = dictionary.doc2bow(text1) # 文章一中所有词在词表中的出现频率 

39 cosine_sim = similarities[test_corpus][1] # 对文章一进行相似度查询 

40 cosine_sim = float("%.6f" % cosine_sim) 

41 return cosine_sim
